3.3 Keyboard Navigation for Sidebar
Update Sidebar.tsx:
tsx// Add keyboard handler
const handleKeyDown = (e: React.KeyboardEvent, item: NavItem) => {
  if (e.key === 'Enter' || e.key === ' ') {
    e.preventDefault();
    if (!isItemLocked(item)) {
      window.location.href = item.path;
    }
  }
};

// Update nav items
<a 
  className={styles.navItem}
  onClick={(e) => handleLockedClick(e, item)}
  onKeyDown={(e) => handleKeyDown(e, item)}
  tabIndex={0}
  role="menuitem"
  aria-disabled={isItemLocked(item)}
>
Add to Sidebar.module.css:
css.navItem:focus-visible {

}
