3. WIRE UP WHAT-IF CALCULATOR BUTTONS

The scenario/calculator buttons need to call the API. Add this function:

const runScenario = async (scenarioType: string) => {
  if (isCalculating) return;
  setIsCalculating(true);
  
  try {
    const response = await fetch('/api/tax-intel/run-scenario', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({ scenarioType, inputs: scenarioInputs }),
    });
    
    if (response.ok) {
      const data = await response.json();
      setScenarios(prev => [...prev, data.scenario]);
      showSuccess('Scenario calculated!');
    } else {
      showError('Failed to run scenario');
    }
  } catch (err) {
    console.error('Scenario error:', err);
    showError('Failed to run scenario');
  } finally {
    setIsCalculating(false);
  }
};

Make each calculator button call this:
<button onClick={() => runScenario('retirement_contribution')}>Calculate 401k Impact</button>
<button onClick={() => runScenario('income_change')}>Calculate Income Change</button>
