2.1 Update ChargeTaxIntel.tsx to use Toast notifications
Add the useToast hook import and implement notifications:
tsx// Add import at top
import { useToast } from '../components/Toast';

// Inside the component, add:
const { showSuccess, showError } = useToast();

// Update uploadAndAnalyze function:
const uploadAndAnalyze = async () => {
  if (!file) return;

  setIsUploading(true);
  try {
    const formData = new FormData();
    formData.append('document', file);
    formData.append('documentType', selectedDocType);
    formData.append('documentYear', String(new Date().getFullYear() - 1));

    const uploadResponse = await fetch('/api/tax-intel/upload', {
      method: 'POST',
      credentials: 'include',
      body: formData,
    });

    if (!uploadResponse.ok) {
      const errorData = await uploadResponse.json();
      throw new Error(errorData.message || 'Upload failed');
    }

    // Show success toast
    showSuccess(`${file.name} uploaded successfully! Analyzing your document...`);

    await loadUploadedDocs();

    const analyzeResponse = await fetch('/api/tax-intel/analyze', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      credentials: 'include',
      body: JSON.stringify({ filename: file.name, documentType: selectedDocType }),
    });

    if (analyzeResponse.ok) {
      const data = await analyzeResponse.json();
      setTaxData(data.taxData);
      showSuccess('Document analyzed! Your tax data is now available.');
    } else {
      showError('Document uploaded but analysis failed. Please try again.');
    }

    setFile(null);
    setSelectedDocType('1040');
  } catch (err: any) {
    console.error('Upload error:', err);
    showError(err.message || 'Failed to upload document. Please try again.', {
      label: 'Retry',
      onClick: () => uploadAndAnalyze()
    });
  } finally {
    setIsUploading(false);
  }
};
